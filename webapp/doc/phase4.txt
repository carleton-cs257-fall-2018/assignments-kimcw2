SQL CREATE STATEMENTS
CREATE TABLE countries (
	id	SERIAL,
	name		TEXT
);

CREATE TABLE varieties (
	id	SERIAL,
	name		TEXT
);

CREATE TABLE wineries (
	id	SERIAL,
	country_id	INT,
	province	TEXT,
	region		TEXT,
	name		TEXT	
);

CREATE TABLE wines (
	description		TEXT,
	designation		TEXT,
	points			INT,
	price			INT,
	taster_name		TEXT,
	taster_twitter_handle	TEXT,
	title 			TEXT,
	variety_id 		INT,
	winery_id		INT
);
